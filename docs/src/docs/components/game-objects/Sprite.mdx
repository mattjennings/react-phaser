import { Playground, Props } from 'docz'
import { Game, Scene, Sprite } from 'react-phaser-fiber'

Sprites are Images with the ability to animate.

<Playground>
  <Game width={400} height={400}>
    <Scene
      sceneKey="main"
      // load the spritesheet for the Sprite
      onPreload={(scene) => {
        scene.load.spritesheet(
          'adventurer',
          '/images/adventurer-spritesheet.png',
          {
            frameWidth: 50,
            frameHeight: 37,
          }
        )
      }}
      // now that the spritesheet is loaded, we'll make the animations.
      onCreate={(scene) => {
        scene.anims.create({
          key: 'idle',
          frames: scene.anims.generateFrameNumbers('adventurer', {
            start: 0,
            end: 3,
          }),
          frameRate: 10,
          repeat: -1,
        })
      }}
    >
      <Sprite x={200} y={200} texture="adventurer" animation="idle" scale={2} />
    </Scene>
  </Game>
</Playground>
